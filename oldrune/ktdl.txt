%ktdl  ^<  "ketdel"

define:
    [%ktdl p=(gene) q=(gene)]

direct:
    [%ktdl *]
  =-  [(nice (fine %very (play p.gen) p.ryd)) q.ryd]
  ^=  ryd  =-  $(gen hos, gol %noun)
  ^=  hos
  :+  %lcps  [%ktlc %a p.gen]                   ::  =+  a=<p.gen>
  :+  %lcps  [%ktlc %b [%lcld [~ 2] q.gen]]     ::  =+  b==>(.2 <q.gen>)
  :+  %csld  [%mtsp %a %b ~]                    ::  ?>  (a b)
  %b                                            ::  b
